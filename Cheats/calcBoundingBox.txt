	@Override
	public void calcBoundingBox() {
	    for(int i = 0; i < 3; i++) {
		min[i] = center[i]-radius;
		max[i] = center[i]+radius;
	    }
	}


	@Override
	public void calcBoundingBox() {
	    min[0] = vertices[0][0];
	    min[1] = vertices[0][1];
	    min[2] = vertices[0][2];
	    max[0] = vertices[0][0];
	    max[1] = vertices[0][1];
	    max[2] = vertices[0][2];
	    for (int i = 1; i < vertices.length; ++i) {
		for (int j = 0; j < 3; j++) {
		    if (min[j] > vertices[i][j])
			min[j] = vertices[i][j];
		    if (max[j] < vertices[i][j])
			max[j] = vertices[i][j];
		}
	    }
	}
